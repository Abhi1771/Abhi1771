#include<stdio.h>
void main()
{
    char str[100], revstr[100];
    int len, i;
    printf("Enter a string:\n");
    fgets(str, 100, stdin);
    while(str[len] != '\0')
    {
        len++;
    }
    for(i=0; i<len; i++)
    {
        revstr[i] = str[len-1-i];
    }
    printf("The reverse of the entered string is:%s", revstr);
}
